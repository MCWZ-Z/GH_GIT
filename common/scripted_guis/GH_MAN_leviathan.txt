scripted_gui = {
	GH_MAN_Map_OpenButtonContainer = {
		window_name = "GH_MAN_leviathan_AccessButton"
		context_type = player_context
		parent_window_token = politics_tab
	
		visible = {
			tag = MAN
		}
	
		effects = {
			GH_MAN_leviathan_OpenButton_click = {
				if = {
					limit = { NOT = { has_country_flag = GH_MAN_leviathan_GUI_Open } }
					set_country_flag = GH_MAN_leviathan_GUI_Open
				}
				else_if = {
					limit = { has_country_flag = GH_MAN_leviathan_GUI_Open }
					clr_country_flag = GH_MAN_Map_GUI_Open
				}
				clr_country_flag = gui_alert_visible
			}
		}
	
		triggers = {
			gui_alert_visible = {
				has_country_flag = gui_alert_visible
			}
		}
	}
}
